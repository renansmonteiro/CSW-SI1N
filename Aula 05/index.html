import numpy as np
    import pandas as pd
    
    # Exemplo de dados (você pode substituir por seu próprio DataFrame)
    dados = pd.DataFrame({
        'valor': [10, 12, 14, 15, 100, 16, 18, 19, 25, 1000, 20, 21, 22]
    })
    
    # Função para detectar e remover outliers usando o IQR
    def remover_outliers(df, coluna):
        # Calculando o primeiro quartil (Q1) e o terceiro quartil (Q3)
        Q1 = df[coluna].quantile(0.25)
        Q3 = df[coluna].quantile(0.75)
        
        # Calculando o intervalo interquartil (IQR)
        IQR = Q3 - Q1
        
        # Definindo os limites inferior e superior
        limite_inferior = Q1 - 1.5 * IQR
        limite_superior = Q3 + 1.5 * IQR
        
        # Removendo os outliers
        df_sem_outliers = df[(df[coluna] >= limite_inferior) & (df[coluna] <= limite_superior)]
        
        return df_sem_outliers
    
    # Aplicando a função para remover outliers
    dados_sem_outliers = remover_outliers(dados, 'valor')
    
    # Exibindo os dados antes e depois de remover os outliers
    print("Dados originais:")
    print(dados)
    
    print("\nDados sem outliers:")
    print(dados_sem_outliers)
